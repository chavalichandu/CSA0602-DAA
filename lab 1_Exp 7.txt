#include <stdio.h>

int isPresent(int arr[], int size, int val) {
    for (int i = 0; i < size; i++)
        if (arr[i] == val)
            return 1;
    return 0;
}

int main() {
    int nums[] = {3, 7, 3, 5, 2, 5, 9, 2};   // Change input here for other test cases
    int n = 8, unique[50], ucount = 0;

    for (int i = 0; i < n; i++) {
        if (!isPresent(unique, ucount, nums[i])) {
            unique[ucount++] = nums[i];
        }
    }

    printf("Unique Elements: [");
    for (int i = 0; i < ucount; i++) {
        printf("%d", unique[i]);
        if (i < ucount - 1) printf(", ");
    }
    printf("]\n");

    return 0;
}
