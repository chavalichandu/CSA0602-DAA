#include <stdio.h>

// Function for Binary Search
int binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1, mid;

    while (low <= high) {
        mid = (low + high) / 2;

        if (arr[mid] == key)
            return mid;     // Element found
        else if (arr[mid] < key)
            low = mid + 1;
        else
            high = mid - 1;
    }
    return -1;   // Not found
}

int main() {
    int arr[] = {-9, 3, 4, 6, 8, 9, 10, 30};  // Must be sorted
    int n = 8;
    int key = 10;  // Change this for other test cases

    int pos = binarySearch(arr, n, key);

    if (pos != -1)
        printf("Element %d is found at position %d\n", key, pos + 1);
    else
        printf("Element %d is not found\n", key);

    return 0;
}
