#include <stdio.h>
#include <string.h>

int isPalindrome(char str[]) {
    int i = 0, j = strlen(str) - 1;
    while (i < j) {
        if (str[i] != str[j])
            return 0;
        i++; j--;
    }
    return 1;
}

int main() {
    char words[5][20] = {"abc", "car", "ada", "racecar", "cool"};
    int n = 5, found = 0;

    for (int i = 0; i < n; i++) {
        if (isPalindrome(words[i])) {
            printf("First Palindromic String: %s\n", words[i]);
            found = 1;
            break;
        }
    }
    if (!found)
        printf("No Palindromic String Found\n");

    return 0;
}
